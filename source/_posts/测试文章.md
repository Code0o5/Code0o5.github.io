void OnFloorIndexPanelStateConfig(bool show)
    {
        if (buildingObj && floorIndexPanelObj) {
            for (int i = 0; i < buildingObj.transform.childCount; i++) {
                Transform child = buildingObj.transform.GetChild(i);
                BuildingFloorSplitFloorInfo floorInfo = child.GetComponent<BuildingFloorSplitFloorInfo>();
                if (floorInfo && floorInfo.floorIndexPanelObj) {
                    Vector3 offsetPos = i * new Vector3(0, 10f, 0);
                    if (show)
                    {
                        floorInfo.floorIndexPanelObj.transform.position = floorInfo.floorIndexPanelOriginPosition + offsetPos;
                        floorIndexPanelObj.SetActive(show);
                    }
                    else
                    {
                        floorIndexPanelObj.SetActive(show);
                        floorInfo.floorIndexPanelObj.transform.position = floorInfo.floorIndexPanelOriginPosition - offsetPos;
                    }
                }
            }
        }
    }
